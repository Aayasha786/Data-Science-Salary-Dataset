#program to show summary statistics of sum, mean, standard deviation, skewness, and kurtosis of any chosen variable
def summary_statistics(df, variable):
    
    # Calculate sum
    Total_sum = df[variable].sum()

    # Calculate mean
    Mean_value = df[variable].mean()

    # Calculate standard deviation
    Standard_dev = df[variable].std()

    # Calculate skewness
    Skewness = df[variable].skew()

    # Calculate kurtosis
    Kurtosis = df[variable].kurt()

    # Print summary statistics
    print(f"Summary Statistics for '{variable}':")
    print(f"Sum: {Total_sum}")
    print(f"Mean: {Mean_value}")
    print(f"Standard Deviation: {Standard_dev}") 
    print(f"Skewness: {Skewness}")  
    print(f"Kurtosis: {Kurtosis}")  


chosen_variables = ['salary_in_usd', 'work_year']
for variable in chosen_variables:
    summary_statistics(df, variable)
    print()

def highest_salary_job(df):
    # Calculate average salary for each job title
    job_salary = df.groupby('job_title')['salary_in_usd'].mean()

    # Find job with highest average salary
    highest_salary_job = job_salary.idxmax()
    highest_salary = job_salary.max()
    plt.figure(figsize=(16,9))

    # Plot bar graph for highest salary
    plt.bar(job_salary.index, job_salary.values, color='skyblue')
    
    # Highlight job with highest salary
    plt.bar(highest_salary_job, highest_salary, color='blue')
    
    # Add horizontal line for highest salary
    plt.title('Job with Highest Salary')
    plt.xlabel('Job Title')
    plt.ylabel('Salary')
    plt.xticks(rotation=45, ha='right',fontsize=6)  # Rotate x-axis labels for better visibility
    plt.show()
